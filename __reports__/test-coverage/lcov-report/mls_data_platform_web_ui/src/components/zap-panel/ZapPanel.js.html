  // @flow
 
  import React, { Component, Fragment } from 'react';
  import { connect } from 'react-redux';
  import { NavLink} from 'react-router-dom';
  import {
    Navbar,
    Nav,
    NavItem,
    Col
  } from 'reactstrap';
  import _ from 'lodash';
  import './ZapPanel.scss';
 
  const mapStateToProps = (state) => {
    return {
      MlsZapPanelData: _.get(state, 'MLSZapPanelData', {}),
      showFullHeader: _.get(state.MLSLoginData, 'showFullHeader', {})
    }
  }
 
  type Props = {
    MlsZapPanelData: {
      panelIsOpen: boolean,
      pathname: string,
    },
    Links: Array<Object>,
    history: Object,
    showFullHeader: {
      isShowFullHeader: boolean,
    }
  };
 
  type State = {
    selectedIndex: number,
  }
 
  const links = [
    {
      name: 'Search MLS',
      icon: 'fa fa-search panelIcons',
      link: '/mlsSearch'
    },
    {
      name: 'Configure MLS',
      icon: 'fa fa-database panelIcons',
      link: '/home'
    },
    {
      name: 'Build MLS',
      icon: 'fa fa-cog panelIcons',
      link: '/urlBuilder'
    },
    {
      name: 'Canonical MLS',
      icon: 'fa fa-sitemap panelIcons',
      link: '/canonicalMappings'
    },
    {
      name: 'Scheduler',
      icon: 'fa fa-calendar panelIcons',
      link: '/home1'
    },
    {
      name: 'Standard Value',
      icon: 'fa fa-exchange panelIcons',
      link: '/standardValue'
    }
  ];
 
  class ZapPanel extends Component<Props, State> {
 
    constructor(props) {
      super(props);
      this.state = {
        selectedIndex: 0,
      }
    }
 
    handleLinkClick = (index) => {
      this.setState({
        selectedIndex: index,
      })
    }
 
    componentDidUpdate(prevProps) {
      const { MlsZapPanelData: { pathname } } = this.props;
      if(prevProps.MlsZapPanelData.pathname !== pathname) {
        this.setState({
          selectedIndex: _.findIndex(links, {link: pathname})
        })
      }
    }
 
    render() {
      const {MlsZapPanelData: { panelIsOpen }, showFullHeader: { isShowFullHeader } } = this.props;
      const { selectedIndex } = this.state;
 
      return (
        <Fragment>
          {isShowFullHeader && <Col xs="6" md="2" className="zapPanel">
            <div>
              <Navbar color="faded" light>
                <Nav navbar>
                { _.map(links, (item, index) => {
                    return (
                      <NavItem key={index} className={selectedIndex === index ? 'selected' : ''}>
                        <div className="nav-item-section">
                            <i className={item.icon}></i>
                            {panelIsOpen &&
                              <NavLink
                                className='linkText'
                                to={{pathname:item.link, state: { selectedIndex: index}}}
                                onClick={() => {
                                  this.handleLinkClick(index)
                                }}
                              >
                                {item.name}
                              </NavLink>}
                        </div>
                      </NavItem>
                      )
                    })
                  }
                </Nav>
              </Navbar>
            </div>
          </Col>}
        </Fragment>
      )
    }
}
 
export default connect(mapStateToProps, null)(ZapPanel);
