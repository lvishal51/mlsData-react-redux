// @flow
 
import React, { Component } from 'react';
import {
  FormGroup,
  Input
} from 'reactstrap';
import _ from 'lodash';
import { connect } from 'react-redux';
import { bindActionCreators } from 'redux';
 
import './AutoComplete.scss';
 
import * as MlsCanonicalActions from 'actions/MlsCanonicalMappingAction';
 
 
const mapDispatchToProps = (dispatch) => ({
  MlsCanonicalActions: bindActionCreators(MlsCanonicalActions, dispatch),
});
 
const mapStateToProps = state => {
  return {    
    MlsGetAutoCompleteOptionsData: _.get(state.MLSCanonicalData, 'MlsAutoCompleteOptions', {}),
  }
}
 
type Props = {
  MlsCanonicalActions: typeof MlsCanonicalActions,
  MlsGetAutoCompleteOptionsData: Object
};
 
type State = {
  availableOptions: Array<Object>,
  autocomplete: string,
  tempItem: Object
};
 
export class AutoComplete extends Component<Props, State> {
 
  constructor(props: Props) {
    super(props);
    this.state = {
      availableOptions: [],
      autocomplete: '',
      tempItem: {}
    }
 
  }
 
  componentDidUpdate = (prevProps: Object) => {
    if(this.props.MlsGetAutoCompleteOptionsData.isResponseSuccess !==  prevProps.MlsGetAutoCompleteOptionsData.isResponseSuccess
      && this.props.MlsGetAutoCompleteOptionsData.isResponseSuccess) {
        const { payload } = this.props.MlsGetAutoCompleteOptionsData;
        this.setState(() => ({ availableOptions: payload }));
      }
  }
 
  handleInputChange = (event: Object ) => {
    this.props.MlsCanonicalActions.getAutoCompleteOptions(event.target.value);
    if (/\)$/g.test(event.target.value)) {
      document.querySelector('.autocomplete__option__info').classList.remove('show');
    }
    this.setState({autocomplete: event.target.value});
  }
 
  completeFunction = (item) => {
    let count = (this.state.autocomplete.match(/[\(\)]/g) || []).length;
    let autocomplete = document.getElementById('autocomplete');
    if (count < 1) {
      this.setState({ autocomplete: item.name + '(', tempItem: item });
    } else {
      let replacement = autocomplete.value.lastIndexOf('(');
      let leftSide = autocomplete.value.substring(0 , replacement+1);
      let rigthSide = autocomplete.value.substring(autocomplete.selectionStart);
      this.setState({ autocomplete: leftSide + item.name + '(' + rigthSide, tempItem: item });
    }
    autocomplete.focus();
    document.querySelector('.autocomplete__container').classList.remove('show');
    document.querySelector('.autocomplete__option__info').classList.add('show');
  }
 
  handleKeyUp = (event: Object) => {
    const autocompleteContainer = document.querySelector('.autocomplete__container');
    const autocompleteInfo = document.querySelector('.autocomplete__option__info');
    if (event.keyCode === 40) { // arrow down
      event.preventDefault();
      autocompleteContainer.firstChild.focus();
      this.setState({ tempItem: this.state.availableOptions[0] });
      autocompleteInfo.classList.add('show');
    } else if (event.keyCode === 38) { //arrow up
      event.preventDefault();
      autocompleteContainer.lastChild.focus();
      this.setState({ tempItem: this.state.availableOptions[this.state.availableOptions.length - 1] });
      autocompleteInfo.classList.add('show');
    }else if (event.keyCode === 27) { //escape key
      autocompleteContainer.classList.remove('show');
      autocompleteInfo.classList.remove('show');
    }else if(event.keyCode === 32 && event.ctrlKey) {
      autocompleteContainer.classList.add('show');
    }
  }
 
  handleKeyUpOption = (index: Number, event: Object) => {
    const autocompleteInfo = document.querySelector('.autocomplete__option__info');
    if (event.keyCode === 40) { // arrow down
      event.preventDefault();
      if(event.target.nextSibling) {
        event.target.nextSibling.focus();
        this.setState({ tempItem: this.state.availableOptions[index + 1] });
        autocompleteInfo.classList.add('show');
      } else {
        event.target.parentNode.firstChild.focus();
        this.setState({ tempItem: this.state.availableOptions[0] });
      }
    } else if (event.keyCode === 38) { //arrow up
      event.preventDefault();
      if (event.target.previousSibling) {
        event.target.previousSibling.focus();
        this.setState({ tempItem: this.state.availableOptions[index - 1] });
        autocompleteInfo.classList.add('show');
      } else {
        event.target.parentNode.lastChild.focus();
        this.setState({ tempItem: this.state.availableOptions[this.state.availableOptions.length - 1] });
      }
    } else if (event.keyCode === 27) { // escape keyp
      event.preventDefault();
      //hide element
      event.target.parentNode.classList.remove('show');
      autocompleteInfo.classList.remove('show');
      document.getElementById('autocomplete').focus();
    } else if (event.keyCode === 13) { // enter key select option
      event.preventDefault();
      this.completeFunction(this.state.availableOptions[index]);
    }
  }
 
  render() {
 
    const { availableOptions, autocomplete, tempItem } = this.state;
 
    return (
      <FormGroup className="position-relative">
        <Input
          type="text"
          name="autocomplete"
          className="form-control"
          id="autocomplete"
          onChange={this.handleInputChange}
          onKeyUp={this.handleKeyUp}
          value={autocomplete}
          />
        <ul className="autocomplete__container" id="listBox" role="listbox">
          {
            _.map(availableOptions,(item, index) => {
              return <li
                        key={index}
                        tabIndex="-1"
                        role="option"
                        onClick={() => this.completeFunction(item)}
                        onKeyUp={(event) => this.handleKeyUpOption(index, event)}
                        className="autocomplete__option">
                        {item.name}
                      </li>
            })
          }
        </ul>
        <div className="autocomplete__option__info">
          <span className="autocomplete__option__info--type">
            {tempItem.type}
          </span>
          <span className="autocomplete__option__info--doc">
            {tempItem.doc}
          </span>
        </div>
      </FormGroup>
    );
  }
}
 
export default connect(mapStateToProps, mapDispatchToProps)(AutoComplete);
